/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
const combine = (n, k) => {
  const result = [];

  /**
   *
   * @param {number[]} current еҪ“еүҚз”ҹжҲҗж•°з»„
   * @param {number[]} rest еү©дҪҷеҸҜеҠ е…Ҙе…ғзҙ 
   */
  const dfs = (current, rest) => {
    if (current.length === k) {
      result.push(current);
      return;
    }

    // еүӘжһқ
    if (current.length + rest.length < k) {
      return;
    }

    rest.forEach((item, index) => {
      dfs([...current, item], [...rest.slice(index + 1)]);
    });
  };

  dfs(
    [],
    Array(n)
      .fill()
      .map((_, i) => i + 1),
  );

  return result;
};
